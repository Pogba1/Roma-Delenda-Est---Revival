# Finland gains cores on ugrian land
country_event = {
    id = 550 
    title = "EVTNAM550"
    desc = "EVTDESC550"
    picture = "football"
    trigger = {
        has_country_flag = embraced_ugrians
            any_owned_province = {
                     AND ={
                       is_core = UGR 
                       NOT = { is_core = FIN}  
                    }
    }
}
	mean_time_to_happen = {
		days = 0
	}

    option = {
        name = "EVTOPTA550"
        prestige = 5
        badboy = -5
         any_owned = {
                    limit = { is_core = UGR}
                   add_core = FIN  
              }
    }
}

# Mongolia wins manchu-mongol war

    id = 553
    title = "EVTNAM553"
    desc = "EVTDESC553"
    picture = "form_union_nation"
    trigger = {
        has_country_flag = claimed_siberia
        war = no
        YAK = { exists = yes NOT = {vassal_of = THIS} }
        SIB = { exists = yes NOT = {vassal_of = THIS} }
        MCK = {
          is_disarmed = yes
          truce_with = THIS
          }
}
	mean_time_to_happen = {
		days = 0
	}

    option = {
        name = "EVTOPTA553"
        prestige = 25
        MCK = { release = SIB release = YAK }
        create_vassal = SIB
        create_vassal = YAK
        KOR = { clr_country_flag = mongol_agression}
        set_country_flag = won_siberia
    }
}
